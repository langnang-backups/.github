з¬¬ 2 жӯҘпјҡж·»еҠ е•Ҷе“Ғ

еңЁ HTML дёӯпјҢж·»еҠ дёҖдёӘж–Үжң¬еӯ—ж®өпјҢе№¶дҪҝз”Ё ng-model жҢҮд»Өе°Ҷе…¶з»‘е®ҡеҲ°еә”з”ЁзЁӢеәҸгҖӮ

еңЁжҺ§еҲ¶еҷЁдёӯпјҢеҲӣе»әдёҖдёӘеҗҚдёә addItem зҡ„еҮҪж•°пјҢе№¶дҪҝз”Ё addMe иҫ“е…Ҙеӯ—ж®өзҡ„еҖје°Ҷе•Ҷе“Ғж·»еҠ еҲ° products ж•°з»„дёӯгҖӮ

з¬¬ 3 жӯҘпјҡеҲ йҷӨе•Ҷе“Ғ

жҲ‘д»¬иҝҳеёҢжңӣиғҪеӨҹд»Һиҙӯзү©жё…еҚ•дёӯеҲ йҷӨе•Ҷе“ҒгҖӮ

еңЁжҺ§еҲ¶еҷЁдёӯпјҢеҲӣе»әдёҖдёӘеҗҚдёә removeItem зҡ„еҮҪж•°пјҢиҜҘеҮҪж•°е°ҶжӮЁжғіиҰҒеҲ йҷӨзҡ„йЎ№зӣ®зҡ„зҙўеј•дҪңдёәеҸӮж•°гҖӮ

еңЁ HTML дёӯпјҢдёәжҜҸдёӘйЎ№зӣ®еҲӣе»әдёҖдёӘ <span> е…ғзҙ пјҢе№¶дёәе…¶жҸҗдҫӣдёҖдёӘ ng-click жҢҮд»ӨпјҢиҜҘжҢҮд»ӨдҪҝз”ЁеҪ“еүҚ $index и°ғз”Ё removeItem еҮҪж•°гҖӮ

е®һдҫӢ

<script>
var app = angular.module("myShoppingList", []);

дәІиҮӘиҜ•дёҖиҜ•

з¬¬ 4 жӯҘпјҡй”ҷиҜҜеӨ„зҗҶ

иҜҘеә”з”ЁжңүдёҖдәӣй”ҷиҜҜпјҢдҫӢеҰӮпјҢеҰӮжһңжӮЁе°қиҜ•дёӨж¬Ўж·»еҠ зӣёеҗҢзҡ„е•Ҷе“ҒпјҢиҜҘеә”з”Ёе°ұдјҡеҙ©жәғгҖӮеҸҰеӨ–пјҢд№ҹдёҚеә”иҜҘе…Ғи®ёж·»еҠ з©әе•Ҷе“ҒгҖӮ

жҲ‘д»¬е°ҶйҖҡиҝҮж·»еҠ ж–°е•Ҷе“ҒеүҚжЈҖжҹҘеҖјжқҘи§ЈеҶіиҝҷдёӘй—®йўҳгҖӮ

еңЁ HTML дёӯпјҢжҲ‘д»¬е°Ҷж·»еҠ дёҖдёӘз”ЁдәҺй”ҷиҜҜж¶ҲжҒҜзҡ„е®№еҷЁпјҢе№¶еңЁжңүдәәе°қиҜ•ж·»еҠ зҺ°жңүе•Ҷе“Ғж—¶зј–еҶҷй”ҷиҜҜж¶ҲжҒҜгҖӮ

е®һдҫӢ

иҙӯзү©жё…еҚ•пјҢеҸҜд»ҘеҶҷе…Ҙй”ҷиҜҜж¶ҲжҒҜпјҡ

<script>
var app = angular.module("myShoppingList", []);
app.controller("myCtrl", function($scope) {
  $scope.products = ["зүӣеҘ¶", "йқўеҢ…", "еҘ¶й…Ә"];
  $scope.addItem = function () {
    $scope.errortext = "";
    if (!$scope.addMe) {return;}
    if ($scope.products.indexOf($scope.addMe) == -1) {
      $scope.products.push($scope.addMe);
    } else {
      $scope.errortext = "иҜҘе•Ҷе“Ғе·Із»ҸеңЁжӮЁзҡ„иҙӯзү©жё…еҚ•дёӯгҖӮ";
    }
  }
  $scope.removeItem = function (x) {
    $scope.errortext = "";
    $scope.products.splice(x, 1);
  }
});
</script>

<div ng-app="myShoppingList" ng-controller="myCtrl">
  <ul>
    <li ng-repeat="x in products">
      {{x}}<span ng-click="removeItem($index)">×</span>
    </li>
  </ul>
  <input ng-model="addMe">
  <button ng-click="addItem()">ж·»еҠ </button>
  <p>{{errortext}}</p>
</div>
