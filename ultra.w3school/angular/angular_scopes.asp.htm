AngularJS дҪңз”Ёеҹҹ

  • AngularJS жҺ§еҲ¶еҷЁ
  • AngularJS иҝҮж»ӨеҷЁ

дҪңз”ЁеҹҹпјҲscopeпјүжҳҜ HTMLпјҲи§Ҷеӣҫпјүе’Ң JavaScriptпјҲжҺ§еҲ¶еҷЁпјүд№Ӣй—ҙзҡ„з»‘е®ҡйғЁеҲҶгҖӮ

дҪңз”ЁеҹҹжҳҜе…·жңүеҸҜз”ЁеұһжҖ§е’Ңж–№жі•зҡ„еҜ№иұЎгҖӮ

дҪңз”ЁеҹҹеҜ№и§Ҷеӣҫе’ҢжҺ§еҲ¶еҷЁйғҪжҳҜеҸҜз”Ёзҡ„гҖӮ

еҰӮдҪ•дҪҝз”ЁдҪңз”Ёеҹҹпјҹ

еҪ“дҪ еңЁ AngularJS дёӯеҲӣе»әжҺ§еҲ¶еҷЁж—¶пјҢдҪ е°Ҷ $scope еҜ№иұЎдҪңдёәеҸӮж•°дј йҖ’пјҡ

е®һдҫӢ

еңЁжҺ§еҲ¶еҷЁдёӯеҲӣе»әзҡ„еұһжҖ§пјҢеҸҜд»ҘеңЁи§Ҷеӣҫдёӯеј•з”Ёпјҡ

<div ng-app="myApp" ng-controller="myCtrl">

<h1>{{carname}}</h1>

</div>

<script>
var app = angular.module('myApp', []);

app.controller('myCtrl', function($scope) {
  $scope.carname = "Volvo";
});
</script>

дәІиҮӘиҜ•дёҖиҜ•

еҪ“еңЁжҺ§еҲ¶еҷЁдёӯеҗ‘ $scope еҜ№иұЎж·»еҠ еұһжҖ§ж—¶пјҢи§ҶеӣҫпјҲHTMLпјүеҸҜд»Ҙи®ҝй—®иҝҷдәӣеұһжҖ§гҖӮ

еңЁи§ҶеӣҫдёӯпјҢдҪ дёҚдҪҝз”Ё $scope еүҚзјҖпјҢдҪ еҸӘйңҖеј•з”ЁдёҖдёӘеұһжҖ§еҗҚпјҢжҜ”еҰӮ {{carname}}гҖӮ

зҗҶи§ЈдҪңз”Ёеҹҹ

еҰӮжһңжҲ‘д»¬жҠҠAngularJSеә”з”ЁзЁӢеәҸзңӢдҪңжҳҜз”ұд»ҘдёӢеҮ йғЁеҲҶз»„жҲҗпјҡ

  • и§ҶеӣҫпјҢеҚі HTMLгҖӮ
  • жЁЎеһӢпјҢеҚіеҪ“еүҚи§ҶеӣҫеҸҜз”Ёзҡ„ж•°жҚ®гҖӮ
  • жҺ§еҲ¶еҷЁпјҢеҚіJavaScriptеҮҪж•°пјҢз”ЁдәҺеҲӣе»ә/жӣҙж”№/еҲ йҷӨ/жҺ§еҲ¶ж•°жҚ®гҖӮ

йӮЈд№ҲдҪңз”Ёеҹҹе°ұжҳҜжЁЎеһӢгҖӮ

дҪңз”ЁеҹҹжҳҜдёҖдёӘе…·жңүеұһжҖ§е’Ңж–№жі•зҡ„ JavaScript еҜ№иұЎпјҢиҝҷдәӣеұһжҖ§е’Ңж–№жі•еҜ№и§Ҷеӣҫе’ҢжҺ§еҲ¶еҷЁйғҪжҳҜеҸҜз”Ёзҡ„гҖӮ

е®һдҫӢ

еҰӮжһңдҪ еңЁи§ҶеӣҫдёӯиҝӣиЎҢдәҶжӣҙж”№пјҢжЁЎеһӢе’ҢжҺ§еҲ¶еҷЁе°Ҷдјҡжӣҙж–°пјҡ

<div ng-app="myApp" ng-controller="myCtrl">

<input ng-model="name">

<h1>My name is {{name}}</h1>

</div>

<script>
var app = angular.module('myApp', []);

app.controller('myCtrl', function($scope) {
  $scope.name = "Bill Gates";
});
</script>

дәІиҮӘиҜ•дёҖиҜ•

дәҶи§ЈдҪ зҡ„дҪңз”Ёеҹҹ

еңЁд»»дҪ•ж—¶еҖҷдәҶи§ЈдҪ жӯЈеңЁеӨ„зҗҶзҡ„дҪңз”ЁеҹҹжҳҜеҫҲйҮҚиҰҒзҡ„гҖӮ

еңЁдёҠйқўзҡ„дёӨдёӘдҫӢеӯҗдёӯпјҢеҸӘжңүдёҖдёӘдҪңз”ЁеҹҹпјҢжүҖд»ҘдәҶи§ЈдҪ зҡ„дҪңз”ЁеҹҹдёҚжҳҜй—®йўҳпјҢдҪҶеҜ№дәҺжӣҙеӨ§зҡ„еә”з”ЁзЁӢеәҸпјҢHTML DOMдёӯзҡ„жҹҗдәӣйғЁеҲҶеҸҜиғҪеҸӘиғҪи®ҝй—®жҹҗдәӣдҪңз”ЁеҹҹгҖӮ

е®һдҫӢ

еҪ“еӨ„зҗҶ ng-repeat жҢҮд»Өж—¶пјҢжҜҸж¬ЎйҮҚеӨҚйғҪеҸҜд»Ҙи®ҝй—®еҪ“еүҚзҡ„йҮҚеӨҚеҜ№иұЎпјҡ

<div ng-app="myApp" ng-controller="myCtrl">

<ul>
  <li ng-repeat="x in names">{{x}}</li>
</ul>

</div>

<script>
var app = angular.module('myApp', []);

app.controller('myCtrl', function($scope) {
  $scope.names = ["Emil", "Tobias", "Linus"];
});
</script>

дәІиҮӘиҜ•дёҖиҜ•

жҜҸдёӘ <li> е…ғзҙ йғҪеҸҜд»Ҙи®ҝй—®еҪ“еүҚзҡ„йҮҚеӨҚеҜ№иұЎпјҢеңЁжң¬дҫӢдёӯжҳҜдёҖдёӘеӯ—з¬ҰдёІпјҢйҖҡиҝҮдҪҝз”Ё x жқҘеј•з”ЁгҖӮ

ж №дҪңз”Ёеҹҹ

жүҖжңүеә”з”ЁзЁӢеәҸйғҪжңүдёҖдёӘ $rootScopeпјҢе®ғжҳҜеңЁеҢ…еҗ« ng-app жҢҮд»Өзҡ„ HTML е…ғзҙ дёҠеҲӣе»әзҡ„дҪңз”ЁеҹҹгҖӮ

rootScope еңЁж•ҙдёӘеә”з”ЁзЁӢеәҸдёӯеҸҜз”ЁгҖӮ

еҰӮжһңжҹҗдёӘеҸҳйҮҸеңЁеҪ“еүҚдҪңз”Ёеҹҹе’Ңж №дҪңз”ЁеҹҹпјҲrootScopeпјүдёӯе…·жңүзӣёеҗҢзҡ„еҗҚз§°пјҢеҲҷеә”з”ЁзЁӢеәҸе°ҶдҪҝз”ЁеҪ“еүҚдҪңз”Ёеҹҹдёӯзҡ„еҸҳйҮҸгҖӮ

е®һдҫӢ

жҺ§еҲ¶еҷЁзҡ„дҪңз”Ёеҹҹе’Ң rootScope дёӯйғҪеӯҳеңЁдёҖдёӘеҗҚдёә "color" зҡ„еҸҳйҮҸпјҡ

<body ng-app="myApp">

<p>The rootScope's favorite color:</p>
<h1>{{color}}</h1>

<div ng-controller="myCtrl">
  <p>The scope of the controller's favorite color:</p>
  <h1>{{color}}</h1>
</div>

<p>The rootScope's favorite color is still:</p>
<h1>{{color}}</h1>

<script>
var app = angular.module('myApp', []);
app.run(function($rootScope) {
  $rootScope.color = 'blue';
});
app.controller('myCtrl', function($scope) {
  $scope.color = "red";
});
</script>
</body>
