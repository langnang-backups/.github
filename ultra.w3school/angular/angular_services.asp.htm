AngularJS жңҚеҠЎ

  • AngularJS иҝҮж»ӨеҷЁ
  • AngularJS Http

еңЁ AngularJS дёӯпјҢжӮЁеҸҜд»ҘеҲӣе»әиҮӘе·ұзҡ„жңҚеҠЎпјҢжҲ–дҪҝз”Ёдј—еӨҡеҶ…зҪ®жңҚеҠЎд№ӢдёҖгҖӮ

д»Җд№ҲжҳҜжңҚеҠЎпјҹ

еңЁ AngularJS дёӯпјҢжңҚеҠЎжҳҜеҮҪж•°жҲ–еҜ№иұЎпјҢеҸҜз”ЁдәҺе№¶д»…йҷҗдәҺжӮЁзҡ„ AngularJS еә”з”ЁзЁӢеәҸгҖӮ

AngularJS жңүеӨ§зәҰ 30 дёӘеҶ…зҪ®жңҚеҠЎгҖӮе…¶дёӯд№ӢдёҖжҳҜ $location жңҚеҠЎгҖӮ

$location жңҚеҠЎжңүж–№жі•еҸҜд»Ҙиҝ”еӣһжңүе…іеҪ“еүҚзҪ‘йЎөдҪҚзҪ®зҡ„дҝЎжҒҜпјҡ

е®һдҫӢ

еңЁжҺ§еҲ¶еҷЁдёӯдҪҝз”Ё $location жңҚеҠЎпјҡ

var app = angular.module('myApp', []);
app.controller('customersCtrl', function($scope, $location) {
    $scope.myUrl = $location.absUrl();
});

дәІиҮӘиҜ•дёҖиҜ•

иҜ·жіЁж„ҸпјҢ$location жңҚеҠЎдҪңдёәеҸӮж•°дј йҖ’еҲ°жҺ§еҲ¶еҷЁгҖӮдёәдәҶеңЁжҺ§еҲ¶еҷЁдёӯдҪҝз”ЁиҜҘжңҚеҠЎпјҢеҝ…йЎ»е°Ҷе…¶е®ҡд№үдёәдҫқиө–йЎ№гҖӮ

дёәд»Җд№ҲдҪҝз”ЁжңҚеҠЎпјҹ

еҜ№дәҺи®ёеӨҡжңҚеҠЎпјҢеҰӮ $location жңҚеҠЎпјҢжӮЁдјјд№ҺеҸҜд»ҘдҪҝз”Ё DOM дёӯе·Іжңүзҡ„еҜ№иұЎпјҲдҫӢеҰӮ window.location еҜ№иұЎпјүпјҢдҪҶе®ғдјҡжңүдёҖдәӣйҷҗеҲ¶пјҢиҮіе°‘еҜ№дәҺжӮЁзҡ„ AngularJS еә”з”ЁзЁӢеәҸжқҘиҜҙжҳҜиҝҷж ·гҖӮ

AngularJS дёҚж–ӯзӣ‘зқЈжӮЁзҡ„еә”з”ЁзЁӢеәҸпјҢдёәдәҶдҪҝе…¶жӯЈзЎ®еӨ„зҗҶжӣҙж”№е’ҢдәӢд»¶пјҢAngularJS жӣҙе–ңж¬ўжӮЁдҪҝз”Ё $location жңҚеҠЎиҖҢдёҚжҳҜ window.location еҜ№иұЎгҖӮ

$http жңҚеҠЎ

$http жңҚеҠЎжҳҜ AngularJS еә”з”ЁзЁӢеәҸдёӯжңҖеёёз”Ёзҡ„жңҚеҠЎд№ӢдёҖгҖӮиҜҘжңҚеҠЎеҗ‘жңҚеҠЎеҷЁеҸ‘еҮәиҜ·жұӮпјҢе№¶и®©жӮЁзҡ„еә”з”ЁзЁӢеәҸеӨ„зҗҶе“Қеә”гҖӮ

е®һдҫӢ

дҪҝз”Ё $http жңҚеҠЎд»ҺжңҚеҠЎеҷЁиҜ·жұӮж•°жҚ®пјҡ

var app = angular.module('myApp', []);
app.controller('myCtrl', function($scope, $http) {
  $http.get("welcome.htm").then(function (response) {
    $scope.myWelcome = response.data;
  });
});

дәІиҮӘиҜ•дёҖиҜ•

жӯӨдҫӢжј”зӨәдәҶ $http жңҚеҠЎзҡ„йқһеёёз®ҖеҚ•зҡ„з”Ёжі•гҖӮиҜ·еңЁ AngularJS Http ж•ҷзЁӢдёӯдәҶи§Јжңүе…і $http жңҚеҠЎзҡ„жӣҙеӨҡдҝЎжҒҜгҖӮ

$timeout жңҚеҠЎ

$timeout жңҚеҠЎжҳҜ AngularJS зүҲжң¬зҡ„ window.setTimeout еҮҪж•°гҖӮ

е®һдҫӢ

дёӨз§’еҗҺжҳҫзӨәж–°ж¶ҲжҒҜпјҡ

var app = angular.module('myApp', []);
app.controller('myCtrl', function($scope, $timeout) {
  $scope.myHeader = "Hello World!";
  $timeout(function () {
    $scope.myHeader = "How are you today?";
  }, 2000);
});

дәІиҮӘиҜ•дёҖиҜ•

$interval жңҚеҠЎ

$interval жңҚеҠЎжҳҜ AngularJS зүҲжң¬зҡ„ window.setInterval еҮҪж•°гҖӮ

е®һдҫӢ

жҜҸз§’жҳҫзӨәж—¶й—ҙпјҡ

var app = angular.module('myApp', []);
app.controller('myCtrl', function($scope, $interval) {
  $scope.theTime = new Date().toLocaleTimeString();
  $interval(function () {
    $scope.theTime = new Date().toLocaleTimeString();
  }, 1000);
});

дәІиҮӘиҜ•дёҖиҜ•

еҲӣе»әжӮЁиҮӘе·ұзҡ„жңҚеҠЎ

иҰҒеҲӣе»әиҮӘе·ұзҡ„жңҚеҠЎпјҢиҜ·е°ҶжңҚеҠЎдёҺжЁЎеқ—иҝһжҺҘпјҡ

еҲӣе»әеҗҚдёә hexafy зҡ„жңҚеҠЎпјҡ

app.service('hexafy', function() {
  this.myFunc = function (x) {
    return x.toString(16);
  }
});

иҰҒдҪҝз”ЁжӮЁзҡ„иҮӘе®ҡд№үжңҚеҠЎпјҢиҜ·еңЁе®ҡд№үжҺ§еҲ¶еҷЁж—¶е°Ҷе…¶ж·»еҠ дёәдҫқиө–йЎ№пјҡ

е®һдҫӢ

дҪҝз”ЁеҗҚдёә hexafy зҡ„иҮӘе®ҡд№үжңҚеҠЎе°Ҷж•°еӯ—иҪ¬жҚўдёәеҚҒе…ӯиҝӣеҲ¶ж•°еӯ—пјҡ

app.controller('myCtrl', function($scope, hexafy) {
  $scope.hex = hexafy.myFunc(255);
});

дәІиҮӘиҜ•дёҖиҜ•

еңЁиҝҮж»ӨеҷЁдёӯдҪҝз”ЁиҮӘе®ҡд№үжңҚеҠЎ

еҲӣе»әжңҚеҠЎе№¶е°Ҷе…¶иҝһжҺҘеҲ°жӮЁзҡ„еә”з”ЁзЁӢеәҸеҗҺпјҢжӮЁеҸҜд»ҘеңЁд»»дҪ•жҺ§еҲ¶еҷЁгҖҒжҢҮд»ӨгҖҒиҝҮж»ӨеҷЁжҲ–з”ҡиҮіе…¶д»–жңҚеҠЎдёӯдҪҝз”ЁиҜҘжңҚеҠЎгҖӮ

иҰҒеңЁиҝҮж»ӨеҷЁдёӯдҪҝз”ЁжңҚеҠЎпјҢиҜ·еңЁе®ҡд№үиҝҮж»ӨеҷЁж—¶е°Ҷе…¶ж·»еҠ дёәдҫқиө–йЎ№пјҡ

иҝҮж»ӨеҷЁ myFormat дёӯдҪҝз”Ёзҡ„жңҚеҠЎ hexafyпјҡ

app.filter('myFormat',['hexafy', function(hexafy) {
  return function(x) {
    return hexafy.myFunc(x);
  };
}]);

дәІиҮӘиҜ•дёҖиҜ•

жӮЁеҸҜд»ҘеңЁжҳҫзӨәеҜ№иұЎжҲ–ж•°з»„дёӯзҡ„еҖјж—¶дҪҝз”ЁиҜҘиҝҮж»ӨеҷЁпјҡ

<ul>
  <li ng-repeat="x in counts">{{x | myFormat}}</li>
</ul>
