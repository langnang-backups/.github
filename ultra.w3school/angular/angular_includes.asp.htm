<body ng-app="">

дәІиҮӘиҜ•дёҖиҜ•

еҢ…еҗ« AngularJS д»Јз Ғ

жӮЁдҪҝз”Ё ng-include жҢҮд»ӨеҢ…еҗ«зҡ„ HTML ж–Үд»¶д№ҹеҸҜд»ҘеҢ…еҗ« AngularJS д»Јз Ғпјҡ

myTable.htm: 
<table>
  <tr ng-repeat="x in names">
    <td>{{ x.Name }}</td>
    <td>{{ x.Country }}</td>
  </tr>
</table>

еңЁжӮЁзҡ„зҪ‘йЎөдёӯеҢ…еҗ« "myTable.htm" ж–Үд»¶пјҢжүҖжңү AngularJS д»Јз ҒйғҪе°Ҷиў«жү§иЎҢпјҢеҚідҪҝжҳҜеҢ…еҗ«ж–Үд»¶дёӯзҡ„д»Јз Ғпјҡ

е®һдҫӢ

<body>

<div ng-app="myApp" ng-controller="customersCtrl">
  <div ng-include="'myTable.htm'"></div>
</div>

<script>
var app = angular.module('myApp', []);
app.controller('customersCtrl', function($scope, $http) {
  $http.get("customers.php").then(function (response) {
    $scope.names = response.data.records;
  });
});
</script>
