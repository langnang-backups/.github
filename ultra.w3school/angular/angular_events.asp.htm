  • ng-mousedown

жӮЁеҸҜд»ҘеңЁд»»дҪ• HTML е…ғзҙ дёҠж·»еҠ йј ж ҮдәӢд»¶гҖӮ

е®һдҫӢ

еҪ“йј ж Үз§»еҠЁеҲ° H1 е…ғзҙ дёҠж—¶пјҢеўһеҠ  count еҸҳйҮҸпјҡ

<div ng-app="myApp" ng-controller="myCtrl">

</div>  
<script>  
var app = angular.module('myApp', []);  
app.controller('myCtrl', function($scope) {  
  $scope.count = 0;  
});  
</script>

дәІиҮӘиҜ•дёҖиҜ•

дҪ д№ҹеҸҜд»Ҙеј•з”ЁдёҖдёӘеҮҪж•°пјҡ

е®һдҫӢ

<div ng-app="myApp" ng-controller="myCtrl">

<script>
var app = angular.module('myApp', []);
app.controller('myCtrl', function($scope) {
  $scope.count = 0;
  $scope.myFunction = function() {
    $scope.count++;
  }
});
</script>

еҲҮжҚўпјҢзңҹ/еҒҮ

еҰӮжһңдҪ жғіеңЁзӮ№еҮ»жҢүй’®ж—¶жҳҫзӨәдёҖйғЁеҲҶ HTML д»Јз ҒпјҢе№¶еңЁеҶҚж¬ЎзӮ№еҮ»жҢүй’®ж—¶йҡҗи—ҸпјҢе°ұеғҸдёӢжӢүиҸңеҚ•дёҖж ·пјҢиҜ·дҪҝжҢүй’®иЎЁзҺ°еҫ—еғҸеҲҮжҚўејҖе…ідёҖж ·пјҡ

зӮ№еҮ»жҲ‘

е®һдҫӢ

<div ng-app="myApp" ng-controller="myCtrl">  
  
<button ng-click="myFunc()">зӮ№еҮ»жҲ‘пјҒ</button>  
  
<div ng-show="showMe">  
  <h1>иҸңеҚ•пјҡ</h1>  
  <div>жҠ«иҗЁ</div>  
  <div>ж„ҸеӨ§еҲ©йқў</div>  
  <div>жө·йІң</div>  
</div>  
  
</div>  
<script>  
var app = angular.module('myApp', []);  
app.controller('myCtrl', function($scope) {  
  $scope.showMe = false;  
  $scope.myFunc = function() {  
    $scope.showMe = !$scope.showMe;  
  }  
});  
</script>

дәІиҮӘиҜ•дёҖиҜ•

showMe еҸҳйҮҸејҖе§Ӣж—¶дёәеёғе°”еҖј falseгҖӮ

myFunc еҮҪж•°йҖҡиҝҮдҪҝз”Ё !пјҲйқһпјүиҝҗз®—з¬ҰпјҢе°Ҷ showMe еҸҳйҮҸи®ҫзҪ®дёәдёҺе…¶еҪ“еүҚеҖјзӣёеҸҚзҡ„еҖјгҖӮ

$event еҜ№иұЎ

и°ғз”ЁеҮҪж•°ж—¶еҸҜд»Ҙе°Ҷ $event еҜ№иұЎдҪңдёәеҸӮж•°дј йҖ’гҖӮ

$event еҜ№иұЎеҢ…еҗ«жөҸи§ҲеҷЁзҡ„дәӢд»¶еҜ№иұЎпјҡ

е®һдҫӢ

<div ng-app="myApp" ng-controller="myCtrl">

<h1 ng-mousemove="myFunc($event)">йј ж Үз§»еҲ°жҲ‘дёҠж–№пјҒ</h1>

<p>Coordinates: {{x + ', ' + y}}</p>

</div>
<script>
var app = angular.module('myApp', []);
app.controller('myCtrl', function($scope) {
  $scope.myFunc = function(myE) {
    $scope.x = myE.clientX;
    $scope.y = myE.clientY;
  }
});
</script>
