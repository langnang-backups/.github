AngularJS иҝҮж»ӨеҷЁ

AngularJS жҸҗдҫӣдәҶиҝҮж»ӨеҷЁжқҘиҪ¬жҚўж•°жҚ®пјҡ

  • currency е°Ҷж•°еӯ—ж јејҸеҢ–дёәиҙ§еёҒж јејҸ
  • date е°Ҷж—Ҙжңҹж јејҸеҢ–дёәжҢҮе®ҡзҡ„ж јејҸ
  • filter д»Һж•°з»„дёӯйҖүжӢ©дёҖдёӘеӯҗйӣҶ
  • json е°ҶеҜ№иұЎж јејҸеҢ–дёә JSON еӯ—з¬ҰдёІ
  • limitTo е°Ҷж•°з»„/еӯ—з¬ҰдёІйҷҗеҲ¶дёәжҢҮе®ҡж•°йҮҸзҡ„е…ғзҙ /еӯ—з¬Ұ
  • lowercase е°Ҷеӯ—з¬ҰдёІж јејҸеҢ–дёәе°ҸеҶҷ
  • number е°Ҷж•°еӯ—ж јејҸеҢ–дёәеӯ—з¬ҰдёІ
  • orderBy йҖҡиҝҮиЎЁиҫҫејҸеҜ№ж•°з»„иҝӣиЎҢжҺ’еәҸ
  • uppercase е°Ҷеӯ—з¬ҰдёІж јејҸеҢ–дёәеӨ§еҶҷ

еҗ‘иЎЁиҫҫејҸж·»еҠ иҝҮж»ӨеҷЁ

еҸҜд»ҘдҪҝз”Ёз®ЎйҒ“з¬Ұ | еҗ‘иЎЁиҫҫејҸж·»еҠ иҝҮж»ӨеҷЁпјҢеҗҺи·ҹдёҖдёӘиҝҮж»ӨеҷЁгҖӮ

uppercase иҝҮж»ӨеҷЁе°Ҷеӯ—з¬ҰдёІж јејҸеҢ–дёәеӨ§еҶҷпјҡ

е®һдҫӢ

<div ng-app="myApp" ng-controller="personCtrl">

<p>еҗҚеӯ—жҳҜ {{ lastName | uppercase }}</p>

</div>

дәІиҮӘиҜ•дёҖиҜ•

lowercase иҝҮж»ӨеҷЁе°Ҷеӯ—з¬ҰдёІж јејҸеҢ–дёәе°ҸеҶҷпјҡ

е®һдҫӢ

<div ng-app="myApp" ng-controller="personCtrl">

<p>еҗҚеӯ—жҳҜ {{ lastName | lowercase }}</p>

</div>

дәІиҮӘиҜ•дёҖиҜ•

еҗ‘жҢҮд»Өж·»еҠ иҝҮж»ӨеҷЁ

йҖҡиҝҮдҪҝз”Ёз®ЎйҒ“еӯ—з¬Ұ | еҗҺи·ҹиҝҮж»ӨеҷЁпјҢеҸҜд»Ҙе°ҶиҝҮж»ӨеҷЁж·»еҠ еҲ°жҢҮд»ӨдёӯпјҢдҫӢеҰӮ ng-repeatпјҡ

е®һдҫӢ

дәІиҮӘиҜ•дёҖиҜ•

currency иҝҮж»ӨеҷЁ

currency иҝҮж»ӨеҷЁе°Ҷж•°еӯ—ж јејҸеҢ–дёәиҙ§еёҒпјҡ

е®һдҫӢ

<div ng-app="myApp" ng-controller="costCtrl">

дәІиҮӘиҜ•дёҖиҜ•

жңүе…і currency иҝҮж»ӨеҷЁзҡ„жӣҙеӨҡдҝЎжҒҜпјҢиҜ·еҸӮйҳ…жҲ‘д»¬зҡ„ AngularJS currency иҝҮж»ӨеҷЁеҸӮиҖғжүӢеҶҢгҖӮ

filter иҝҮж»ӨеҷЁ

filter иҝҮж»ӨеҷЁйҖүжӢ©ж•°з»„зҡ„еӯҗйӣҶгҖӮ

filter иҝҮж»ӨеҷЁеҸӘиғҪз”ЁдәҺж•°з»„пјҢе®ғиҝ”еӣһеҸӘеҢ…еҗ«еҢ№й…ҚйЎ№зҡ„ж•°з»„гҖӮ

жңүе…і filter иҝҮж»ӨеҷЁзҡ„жӣҙеӨҡдҝЎжҒҜпјҢиҜ·еҸӮйҳ…жҲ‘д»¬зҡ„ AngularJS filter иҝҮж»ӨеҷЁеҸӮиҖғжүӢеҶҢгҖӮ

ж №жҚ®з”ЁжҲ·иҫ“е…ҘиҝҮж»Өж•°з»„

еңЁиҫ“е…Ҙеӯ—ж®өдёӯиҫ“е…ҘдёҖдёӘеӯ—жҜҚпјҢеҲ—иЎЁе°Ҷж №жҚ®еҢ№й…ҚйЎ№иҝӣиЎҢзј©е°Ҹ/жү©еӨ§пјҡ

йҖҡиҝҮеңЁиЎЁеӨҙж·»еҠ  ng-click жҢҮд»ӨпјҢжҲ‘д»¬еҸҜд»ҘиҝҗиЎҢдёҖдёӘеҮҪж•°пјҢиҜҘеҮҪж•°дјҡжӣҙж”№ж•°з»„зҡ„жҺ’еәҸйЎәеәҸпјҡ

иҮӘе®ҡд№үиҝҮж»ӨеҷЁ

жӮЁеҸҜд»ҘйҖҡиҝҮеҗ‘жЁЎеқ—жіЁеҶҢдёҖдёӘж–°зҡ„иҝҮж»ӨеҷЁе·ҘеҺӮеҮҪж•°жқҘеҲӣе»әиҮӘе·ұзҡ„иҝҮж»ӨеҷЁпјҡ

е®һдҫӢ

еҲӣе»әдёҖдёӘеҗҚдёә "myFormat" зҡ„иҮӘе®ҡд№үиҝҮж»ӨеҷЁпјҡ

<ul ng-app="myApp" ng-controller="namesCtrl">
  <li ng-repeat="x in names">
    {{x | myFormat}}
  </li>
</ul>

<script>
var app = angular.module('myApp', []);
app.filter('myFormat', function() {
  return function(x) {
    var i, c, txt = "";
    for (i = 0; i < x.length; i++) {
      c = x[i];
      if (i % 2 == 0) {
        c = c.toUpperCase();
      }
      txt += c;
    }
    return txt;
  };
});
app.controller('namesCtrl', function($scope) {
  $scope.names = ['Jani', 'Carl', 'Margareth', 'Hege', 'Joe', 'Gustav', 'Birgit', 'Mary', 'Kai'];
});
</script>
