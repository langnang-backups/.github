AUTO INCREMENT еӯ—ж®ө

жҲ‘д»¬йҖҡеёёеёҢжңӣеңЁжҜҸж¬ЎжҸ’е…Ҙж–°и®°еҪ•ж—¶пјҢиҮӘеҠЁең°еҲӣе»әдё»й”®еӯ—ж®өзҡ„еҖјгҖӮ

жҲ‘д»¬еҸҜд»ҘеңЁиЎЁдёӯеҲӣе»әдёҖдёӘ auto-increment еӯ—ж®өгҖӮ

з”ЁдәҺ MySQL зҡ„иҜӯжі•

дёӢеҲ— SQL иҜӯеҸҘжҠҠ "Persons" иЎЁдёӯзҡ„ "P_Id" еҲ—е®ҡд№үдёә auto-increment дё»й”®пјҡ

CREATE TABLE Persons
(
P_Id int NOT NULL AUTO_INCREMENT,
LastName varchar(255) NOT NULL,
FirstName varchar(255),
Address varchar(255),
City varchar(255),
PRIMARY KEY (P_Id)
)

MySQL дҪҝз”Ё AUTO_INCREMENT е…ій”®еӯ—жқҘжү§иЎҢ auto-increment д»»еҠЎгҖӮ

й»ҳи®Өең°пјҢAUTO_INCREMENT зҡ„ејҖе§ӢеҖјжҳҜ 1пјҢжҜҸжқЎж–°и®°еҪ•йҖ’еўһ 1гҖӮ

иҰҒи®© AUTO_INCREMENT еәҸеҲ—д»Ҙе…¶д»–зҡ„еҖјиө·е§ӢпјҢиҜ·дҪҝз”ЁдёӢеҲ— SQL иҜӯжі•пјҡ

ALTER TABLE Persons AUTO_INCREMENT=100

иҰҒеңЁ "Persons" иЎЁдёӯжҸ’е…Ҙж–°и®°еҪ•пјҢжҲ‘д»¬дёҚеҝ…дёә "P_Id" еҲ—и§„е®ҡеҖјпјҲдјҡиҮӘеҠЁж·»еҠ дёҖдёӘе”ҜдёҖзҡ„еҖјпјүпјҡ

INSERT INTO Persons (FirstName,LastName)
VALUES ('Bill','Gates')

дёҠйқўзҡ„ SQL иҜӯеҸҘдјҡеңЁ "Persons" иЎЁдёӯжҸ’е…ҘдёҖжқЎж–°и®°еҪ•гҖӮ"P_Id" дјҡиў«иөӢдәҲдёҖдёӘе”ҜдёҖзҡ„еҖјгҖӮ"FirstName" дјҡиў«и®ҫзҪ®дёә "Bill"пјҢ"LastName" еҲ—дјҡиў«и®ҫзҪ®дёә "Gates"гҖӮ

з”ЁдәҺ SQL Server зҡ„иҜӯжі•

дёӢеҲ— SQL иҜӯеҸҘжҠҠ "Persons" иЎЁдёӯзҡ„ "P_Id" еҲ—е®ҡд№үдёә auto-increment дё»й”®пјҡ

MS SQL дҪҝз”Ё IDENTITY е…ій”®еӯ—жқҘжү§иЎҢ auto-increment д»»еҠЎгҖӮ

й»ҳи®Өең°пјҢIDENTITY зҡ„ејҖе§ӢеҖјжҳҜ 1пјҢжҜҸжқЎж–°и®°еҪ•йҖ’еўһ 1гҖӮ

иҰҒи§„е®ҡ "P_Id" еҲ—д»Ҙ 20 иө·е§Ӣдё”йҖ’еўһ 10пјҢиҜ·жҠҠ identity ж”№дёә IDENTITY(20,10)
