JavaScript еӯ—з¬ҰдёІж–№жі•

  • JS еӯ—з¬ҰдёІ
  • JS еӯ—з¬ҰдёІжҗңзҙў

еӯ—з¬ҰдёІж–№жі•её®еҠ©жӮЁеӨ„зҗҶеӯ—з¬ҰдёІгҖӮ

еӯ—з¬ҰдёІж–№жі•е’ҢеұһжҖ§

еҺҹе§ӢеҖјпјҢжҜ”еҰӮвҖңBill GatesвҖқпјҢж— жі•жӢҘжңүеұһжҖ§е’Ңж–№жі•пјҲеӣ дёәе®ғд»¬дёҚжҳҜеҜ№иұЎпјүгҖӮ

дҪҶжҳҜйҖҡиҝҮ JavaScriptпјҢж–№жі•е’ҢеұһжҖ§д№ҹеҸҜз”ЁдәҺеҺҹе§ӢеҖјпјҢеӣ дёәеңЁжү§иЎҢж–№жі•е’ҢеұһжҖ§ж—¶ JavaScript е°ҶеҺҹе§ӢеҖји§ҶдёәеҜ№иұЎгҖӮ

еӯ—з¬ҰдёІй•ҝеәҰ

length еұһжҖ§иҝ”еӣһеӯ—з¬ҰдёІзҡ„й•ҝеәҰпјҡ

е®һдҫӢ

var txt = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
var sln = txt.length;

дәІиҮӘиҜ•дёҖиҜ•

жҹҘжүҫеӯ—з¬ҰдёІдёӯзҡ„еӯ—з¬ҰдёІ

indexOf() ж–№жі•иҝ”еӣһеӯ—з¬ҰдёІдёӯжҢҮе®ҡж–Үжң¬йҰ–ж¬ЎеҮәзҺ°зҡ„зҙўеј•пјҲдҪҚзҪ®пјүпјҡ

е®һдҫӢ

var str = "The full name of China is the People's Republic of China.";
var pos = str.indexOf("China");

дәІиҮӘиҜ•дёҖиҜ•

JavaScript д»Һйӣ¶и®Ўз®—дҪҚзҪ®гҖӮ

0 жҳҜеӯ—з¬ҰдёІдёӯзҡ„з¬¬дёҖдёӘдҪҚзҪ®пјҢ1 жҳҜз¬¬дәҢдёӘпјҢ2 жҳҜз¬¬дёүдёӘ ...

lastIndexOf() ж–№жі•иҝ”еӣһжҢҮе®ҡж–Үжң¬еңЁеӯ—з¬ҰдёІдёӯжңҖеҗҺдёҖж¬ЎеҮәзҺ°зҡ„зҙўеј•пјҡ

е®һдҫӢ

var str = "The full name of China is the People's Republic of China.";
var pos = str.lastIndexOf("China");

дәІиҮӘиҜ•дёҖиҜ•

еҰӮжһңжңӘжүҫеҲ°ж–Үжң¬пјҢ indexOf() е’Ң lastIndexOf() еқҮиҝ”еӣһ -1гҖӮ

е®һдҫӢ

var str = "The full name of China is the People's Republic of China.";
var pos = str.indexOf("USA");

дәІиҮӘиҜ•дёҖиҜ•

дёӨз§Қж–№жі•йғҪжҺҘеҸ—дҪңдёәжЈҖзҙўиө·е§ӢдҪҚзҪ®зҡ„з¬¬дәҢдёӘеҸӮж•°гҖӮ

е®һдҫӢ

var str = "The full name of China is the People's Republic of China.";
var pos = str.indexOf("China", 18);

дәІиҮӘиҜ•дёҖиҜ•

lastIndexOf() ж–№жі•еҗ‘еҗҺиҝӣиЎҢжЈҖзҙўпјҲд»Һе°ҫеҲ°еӨҙпјүпјҢиҝҷж„Ҹе‘ізқҖпјҡеҒҮеҰӮз¬¬дәҢдёӘеҸӮж•°жҳҜ 50пјҢеҲҷд»ҺдҪҚзҪ® 50 ејҖе§ӢжЈҖзҙўпјҢзӣҙеҲ°еӯ—з¬ҰдёІзҡ„иө·зӮ№гҖӮ

е®һдҫӢ

var str = "The full name of China is the People's Republic of China.";
var pos = str.lastIndexOf("China", 50);

дәІиҮӘиҜ•дёҖиҜ•

жЈҖзҙўеӯ—з¬ҰдёІдёӯзҡ„еӯ—з¬ҰдёІ

search() ж–№жі•жҗңзҙўзү№е®ҡеҖјзҡ„еӯ—з¬ҰдёІпјҢе№¶иҝ”еӣһеҢ№й…Қзҡ„дҪҚзҪ®пјҡ

е®һдҫӢ

var str = "The full name of China is the People's Republic of China.";
var pos = str.search("locate");

дәІиҮӘиҜ•дёҖиҜ•

жӮЁжіЁж„ҸеҲ°дәҶеҗ—пјҹ

дёӨз§Қж–№жі•пјҢindexOf() дёҺ search()пјҢжҳҜзӣёзӯүзҡ„гҖӮ

иҝҷдёӨз§Қж–№жі•жҳҜдёҚзӣёзӯүзҡ„гҖӮеҢәеҲ«еңЁдәҺпјҡ

  • search() ж–№жі•ж— жі•и®ҫзҪ®з¬¬дәҢдёӘејҖе§ӢдҪҚзҪ®еҸӮж•°гҖӮ
  • indexOf() ж–№жі•ж— жі•и®ҫзҪ®жӣҙејәеӨ§зҡ„жҗңзҙўеҖјпјҲжӯЈеҲҷиЎЁиҫҫејҸпјүгҖӮ

жӮЁе°ҶеңЁжӯЈеҲҷиЎЁиҫҫејҸзҡ„з« иҠӮеӯҰд№ еҲ°иҝҷдәӣжӣҙејәеӨ§зҡ„жЈҖзҙўеҖјгҖӮ

жҸҗеҸ–йғЁеҲҶеӯ—з¬ҰдёІ

жңүдёүз§ҚжҸҗеҸ–йғЁеҲҶеӯ—з¬ҰдёІзҡ„ж–№жі•пјҡ

  • slice(start, end)
  • substring(start, end)
  • substr(start, length)

slice() ж–№жі•

slice() жҸҗеҸ–еӯ—з¬ҰдёІзҡ„жҹҗдёӘйғЁеҲҶе№¶еңЁж–°еӯ—з¬ҰдёІдёӯиҝ”еӣһиў«жҸҗеҸ–зҡ„йғЁеҲҶгҖӮ

иҜҘж–№жі•и®ҫзҪ®дёӨдёӘеҸӮж•°пјҡиө·е§Ӣзҙўеј•пјҲејҖе§ӢдҪҚзҪ®пјүпјҢз»Ҳжӯўзҙўеј•пјҲз»“жқҹдҪҚзҪ®пјүгҖӮ

иҝҷдёӘдҫӢеӯҗиЈҒеүӘеӯ—з¬ҰдёІдёӯдҪҚзҪ® 7 еҲ°дҪҚзҪ® 13 зҡ„зүҮж®өпјҡ

е®һдҫӢ

var str = "Apple, Banana, Mango";
var res = str.slice(7,13);

res зҡ„з»“жһңжҳҜпјҡ

Banana

дәІиҮӘиҜ•дёҖиҜ•

еҰӮжһңжҹҗдёӘеҸӮж•°дёәиҙҹпјҢеҲҷд»Һеӯ—з¬ҰдёІзҡ„з»“е°ҫејҖе§Ӣи®Ўж•°гҖӮ

иҝҷдёӘдҫӢеӯҗиЈҒеүӘеӯ—з¬ҰдёІдёӯдҪҚзҪ® -12 еҲ°дҪҚзҪ® -6 зҡ„зүҮж®өпјҡ

е®һдҫӢ

var str = "Apple, Banana, Mango";
var res = str.slice(-13,-7);

res зҡ„з»“жһңжҳҜпјҡ

Banana

дәІиҮӘиҜ•дёҖиҜ•

еҰӮжһңзңҒз•Ҙз¬¬дәҢдёӘеҸӮж•°пјҢеҲҷиҜҘж–№жі•е°ҶиЈҒеүӘеӯ—з¬ҰдёІзҡ„еү©дҪҷйғЁеҲҶпјҡ

е®һдҫӢ

var res = str.slice(7);

дәІиҮӘиҜ•дёҖиҜ•

жҲ–иҖ…д»Һз»“е°ҫи®Ўж•°пјҡ

е®һдҫӢ

var res = str.slice(-13);

дәІиҮӘиҜ•дёҖиҜ•

жҸҗзӨәпјҡиҙҹеҖјдҪҚзҪ®дёҚйҖӮз”Ё Internet Explorer 8 еҸҠе…¶жӣҙж—©зүҲжң¬гҖӮ

substring() зұ»дјјдәҺ slice()гҖӮ

дёҚеҗҢд№ӢеӨ„еңЁдәҺ substring() ж— жі•жҺҘеҸ—иҙҹзҡ„зҙўеј•гҖӮ

е®һдҫӢ

var str = "Apple, Banana, Mango";
var res = str.substring(7,13);

res зҡ„з»“жһңжҳҜпјҡ

Banana

дәІиҮӘиҜ•дёҖиҜ•

еҰӮжһңзңҒз•Ҙз¬¬дәҢдёӘеҸӮж•°пјҢеҲҷиҜҘ substring() е°ҶиЈҒеүӘеӯ—з¬ҰдёІзҡ„еү©дҪҷйғЁеҲҶгҖӮ

substr() ж–№жі•

substr() зұ»дјјдәҺ slice()гҖӮ

дёҚеҗҢд№ӢеӨ„еңЁдәҺз¬¬дәҢдёӘеҸӮж•°и§„е®ҡиў«жҸҗеҸ–йғЁеҲҶзҡ„й•ҝеәҰгҖӮ

е®һдҫӢ

var str = "Apple, Banana, Mango";
var res = str.substr(7,6);

res зҡ„з»“жһңжҳҜпјҡ

Banana

дәІиҮӘиҜ•дёҖиҜ•

еҰӮжһңзңҒз•Ҙз¬¬дәҢдёӘеҸӮж•°пјҢеҲҷиҜҘ substr() е°ҶиЈҒеүӘеӯ—з¬ҰдёІзҡ„еү©дҪҷйғЁеҲҶгҖӮ

е®һдҫӢ

var str = "Apple, Banana, Mango";

дәІиҮӘиҜ•дёҖиҜ•

Banana, Mango

еҰӮжһңйҰ–дёӘеҸӮж•°дёәиҙҹпјҢеҲҷд»Һеӯ—з¬ҰдёІзҡ„з»“е°ҫи®Ўз®—дҪҚзҪ®гҖӮ

var str = "Apple, Banana, Mango";

дәІиҮӘиҜ•дёҖиҜ•

Mango

з¬¬дәҢдёӘеҸӮж•°дёҚиғҪдёәиҙҹпјҢеӣ дёәе®ғе®ҡд№үзҡ„жҳҜй•ҝеәҰгҖӮ

жӣҝжҚўеӯ—з¬ҰдёІеҶ…е®№

replace() ж–№жі•з”ЁеҸҰдёҖдёӘеҖјжӣҝжҚўеңЁеӯ—з¬ҰдёІдёӯжҢҮе®ҡзҡ„еҖјпјҡ

е®һдҫӢ

str = "Please visit Microsoft!";
var n = str.replace("Microsoft", "W3School");

дәІиҮӘиҜ•дёҖиҜ•

replace() ж–№жі•дёҚдјҡж”№еҸҳи°ғз”Ёе®ғзҡ„еӯ—з¬ҰдёІгҖӮе®ғиҝ”еӣһзҡ„жҳҜж–°еӯ—з¬ҰдёІгҖӮ

й»ҳи®Өең°пјҢreplace() еҸӘжӣҝжҚўйҰ–дёӘеҢ№й…Қпјҡ

е®һдҫӢ

str = "Please visit Microsoft and Microsoft!";
var n = str.replace("Microsoft", "W3School");

дәІиҮӘиҜ•дёҖиҜ•

й»ҳи®Өең°пјҢreplace() еҜ№еӨ§е°ҸеҶҷж•Ҹж„ҹгҖӮеӣ жӯӨдёҚеҜ№еҢ№й…Қ MICROSOFTпјҡ

е®һдҫӢ

str = "Please visit Microsoft!";
var n = str.replace("MICROSOFT", "W3School");

дәІиҮӘиҜ•дёҖиҜ•

еҰӮйңҖжү§иЎҢеӨ§е°ҸеҶҷдёҚж•Ҹж„ҹзҡ„жӣҝжҚўпјҢиҜ·дҪҝз”ЁжӯЈеҲҷиЎЁиҫҫејҸ /iпјҲеӨ§е°ҸеҶҷдёҚж•Ҹж„ҹпјүпјҡ

е®һдҫӢ

str = "Please visit Microsoft!";
var n = str.replace(/MICROSOFT/i, "W3School");

дәІиҮӘиҜ•дёҖиҜ•

жӮЁе°ҶеңЁ JavaScript жӯЈеҲҷиЎЁиҫҫејҸиҝҷдёҖз« еӯҰеҲ°жӣҙеӨҡжңүе…іжӯЈеҲҷиЎЁиҫҫејҸзҡ„еҶ…е®№гҖӮ

var text1 = "Hello World!";       // еӯ—з¬ҰдёІ

дәІиҮӘиҜ•дёҖиҜ•

е®һдҫӢ

var text1 = "Hello World!";       // еӯ—з¬ҰдёІ

var text = "Hello" + " " + "World!";
var text = "Hello".concat(" ","World!");

жүҖжңүеӯ—з¬ҰдёІж–№жі•йғҪдјҡиҝ”еӣһж–°еӯ—з¬ҰдёІгҖӮе®ғд»¬дёҚдјҡдҝ®ж”№еҺҹе§Ӣеӯ—з¬ҰдёІгҖӮ

жӯЈејҸең°иҜҙпјҡеӯ—з¬ҰдёІжҳҜдёҚеҸҜеҸҳзҡ„пјҡеӯ—з¬ҰдёІдёҚиғҪжӣҙж”№пјҢеҸӘиғҪжӣҝжҚўгҖӮ

дәІиҮӘиҜ•дёҖиҜ•

еҰӮйңҖж”ҜжҢҒ IE 8пјҢжӮЁеҸҜжҗӯй…ҚжӯЈеҲҷиЎЁиҫҫејҸдҪҝз”Ё replace() ж–№жі•д»Јжӣҝпјҡ

var str = "       Hello World!        ";

дәІиҮӘиҜ•дёҖиҜ•

жӮЁиҝҳеҸҜд»ҘдҪҝз”ЁдёҠйқўзҡ„ replace ж–№жЎҲжҠҠ trim еҮҪж•°ж·»еҠ еҲ° JavaScript String.prototypeпјҡ

};

charAt() ж–№жі•
