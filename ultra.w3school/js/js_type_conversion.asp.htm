дёҖе…ғзҡ„ + иҝҗз®—з¬ҰеҸҜз”ЁдәҺжҠҠеҸҳйҮҸиҪ¬жҚўдёәж•°еӯ—пјҡ

е®һдҫӢ

var y = "5";      // y жҳҜеӯ—з¬ҰдёІ
var x = + y;      // x жҳҜж•°еӯ—

дәІиҮӘиҜ•дёҖиҜ•

еҰӮжһңж— жі•иҪ¬жҚўеҸҳйҮҸпјҢеҲҷд»ҚдјҡжҲҗдёәж•°еӯ—пјҢдҪҶжҳҜеҖјдёә NaNпјҲNot a numberпјүпјҡ

е®һдҫӢ

var y = "Bill";   // y жҳҜеӯ—з¬ҰдёІ
var x = + y;      // x жҳҜж•°еӯ— (NaN)

дәІиҮӘиҜ•дёҖиҜ•

жҠҠеёғе°”иҪ¬жҚўж•°еҖј

е…ЁеұҖж–№жі• Number() д№ҹеҸҜжҠҠеёғе°”иҪ¬жҚўдёәж•°еӯ—гҖӮ

Number(false)     // иҝ”еӣһ 0
Number(true)      // иҝ”еӣһ 1

жҠҠж—ҘжңҹиҪ¬жҚўдёәж•°еӯ—

е…ЁеұҖж–№жі• Number() еҸҜз”ЁдәҺжҠҠж—ҘжңҹиҪ¬жҚўдёәж•°еӯ—гҖӮ

d = new Date();
Number(d)

ж—Ҙжңҹж–№жі• getTime() еҗҢзҗҶгҖӮ

d.getTime()

иҮӘеҠЁзұ»еһӢиҪ¬жҚў

еҰӮжһң JavaScript е°қиҜ•ж“ҚдҪңдёҖз§ҚвҖңй”ҷиҜҜвҖқзҡ„ж•°жҚ®зұ»еһӢпјҢе®ғдјҡиҜ•еӣҫе°ҶиҜҘеҖјиҪ¬жҚўдёәвҖңжӯЈзЎ®вҖқзҡ„зұ»еһӢгҖӮ

з»“жһңе№¶дёҚжҖ»жҳҜдҪ жүҖжңҹжңӣзҡ„пјҡ

5 + null    // иҝ”еӣһ 5         еӣ дёә null иў«иҪ¬жҚўдёә 0
"5" + null  // иҝ”еӣһ "5null"   еӣ дёә null иў«иҪ¬жҚўдёә  "null"
"5" + 2     // иҝ”еӣһ 52        еӣ дёә 2 иў«иҪ¬жҚўдёә "2"
"5" - 2     // иҝ”еӣһ 3         еӣ дёә "5" иў«иҪ¬жҚўдёә 5
"5" * "2"   // иҝ”еӣһ 10        еӣ дёә "5" е’Ң "2" иў«иҪ¬жҚўдёә 5 е’Ң 2

иҮӘеҠЁеӯ—з¬ҰдёІиҪ¬жҚў
