XSLT - е®ўжҲ·з«Ҝ

  • XSLT apply
  • XSLT еңЁжңҚеҠЎеҷЁз«Ҝ

еҰӮжһңжӮЁзҡ„жөҸи§ҲеҷЁж”ҜжҢҒ XSLTпјҢйӮЈд№ҲеңЁжөҸи§ҲеҷЁдёӯе®ғеҸҜиў«з”ЁжқҘе°Ҷж–ҮжЎЈиҪ¬жҚўдёә XHTMLгҖӮ

JavaScript и§ЈеҶіж–№жЎҲ

еңЁеүҚйқўзҡ„з« иҠӮпјҢжҲ‘д»¬е·Іеҗ‘жӮЁи®Іи§ЈеҰӮдҪ•дҪҝз”Ё XSLT е°ҶжҹҗдёӘ XML ж–ҮжЎЈиҪ¬жҚўдёә XHTMLгҖӮжҲ‘д»¬жҳҜйҖҡиҝҮд»ҘдёӢйҖ”еҫ„е®ҢжҲҗиҝҷдёӘе·ҘдҪңзҡ„пјҡеҗ‘ XML ж–Үд»¶ж·»еҠ  XSL ж ·ејҸиЎЁпјҢе№¶йҖҡиҝҮжөҸи§ҲеҷЁе®ҢжҲҗиҪ¬жҚўгҖӮ

еҚідҪҝиҝҷз§Қж–№жі•зҡ„ж•ҲжһңеҫҲеҘҪпјҢеңЁ XML ж–Үд»¶дёӯеҢ…еҗ«ж ·ејҸиЎЁеј•з”Ёд№ҹдёҚжҖ»жҳҜд»Өдәәж»Ўж„Ҹзҡ„пјҲдҫӢеҰӮпјҢеңЁж— жі•иҜҶеҲ«XSLTзҡ„жөҸи§ҲеҷЁиҝҷз§Қж–№жі•е°ұж— жі•еҘҸж•ҲпјүгҖӮ

жӣҙйҖҡз”Ёзҡ„ж–№жі•жҳҜдҪҝз”Ё JavaScript жқҘе®ҢжҲҗиҪ¬жҚўгҖӮ

йҖҡиҝҮдҪҝз”Ё JavaScriptпјҢжҲ‘д»¬еҸҜд»Ҙпјҡ

  • иҝӣиЎҢжөҸи§ҲеҷЁзЎ®и®ӨжөӢиҜ•
  • ж №жҚ®жөҸи§ҲеҷЁе’ҢдҪҝз”ЁиҖ…зҡ„йңҖжұӮжқҘдҪҝз”ЁдёҚеҗҢзҡ„ж ·ејҸиЎЁ

иҝҷе°ұжҳҜ XSLT зҡ„йӯ…еҠӣжүҖеңЁпјҒXSLT зҡ„и®ҫи®Ўзӣ®зҡ„д№ӢдёҖе°ұжҳҜдҪҝдёҖз§Қж јејҸеҲ°еҸҰдёҖз§Қж јејҸзҡ„иҪ¬жҚўжҲҗдёәеҸҜиғҪпјҢеҗҢж—¶ж”ҜжҢҒдёҚеҗҢзұ»еһӢзҡ„жөҸи§ҲеҷЁд»ҘеҸҠдёҚеҗҢзҡ„з”ЁжҲ·йңҖжұӮгҖӮ

жөҸи§ҲеҷЁз«Ҝзҡ„ XSLT иҪ¬жҚўдёҖе®ҡдјҡжҲҗдёәжңӘжқҘжөҸи§ҲеҷЁжүҖжү§иЎҢзҡ„дё»иҰҒд»»еҠЎд№ӢдёҖпјҢеҗҢж—¶жҲ‘д»¬д№ҹдјҡзңӢеҲ°е…¶еңЁзү№е®ҡзҡ„жөҸи§ҲеҷЁеёӮеңәзҡ„еўһй•ҝпјҲзӣІж–ҮгҖҒзҪ‘з»ңжү“еҚ°жңәпјҢеҗ¬и§үи®ҫеӨҮпјҢзӯүзӯүпјүгҖӮ

XML ж–Үд»¶е’Ң XSL ж–Үд»¶

иҜ·зңӢиҝҷдёӘеңЁеүҚйқўзҡ„з« иҠӮе·Іеұ•зӨәиҝҮзҡ„ XML ж–ҮжЎЈпјҡ

<?xml version="1.0" encoding="ISO-8859-1"?>
<catalog>
  <cd>
    <title>Empire Burlesque</title>
    <artist>Bob Dylan</artist>
    <country>USA</country>
    <company>Columbia</company>
    <price>10.90</price>
    <year>1985</year>
  </cd>
.
.
.
</catalog>

жҹҘзңӢжӯӨ XML ж–Үд»¶гҖӮ

д»ҘеҸҠйҷ„йҡҸзҡ„ XSL ж ·ејҸиЎЁпјҡ

<?xml version="1.0" encoding="ISO-8859-1"?>
<xsl:stylesheet version="1.0"
xmlns:xsl="http://www.w3.org/1999/XSL/Transform">

<xsl:template match="/">
  <html>
  <body>
    <h2>My CD Collection</h2> 
    <table border="1">
      <tr bgcolor="#9acd32">
        <th align="left">Title</th> 
        <th align="left">Artist</th> 
      </tr>
      <xsl:for-each select="catalog/cd">
      <tr>
        <td><xsl:value-of select="title" /></td>
        <td><xsl:value-of select="artist" /></td>
      </tr>
      </xsl:for-each>
  </table>
  </body>
  </html>
</xsl:template>

</xsl:stylesheet>

жҹҘзңӢжӯӨ XSL ж–Үд»¶гҖӮ

иҜ·жіЁж„ҸпјҢиҝҷдёӘ XML ж–Үд»¶жІЎжңүеҢ…еҗ«еҜ№ XSL ж–Үд»¶зҡ„еј•з”ЁгҖӮ

йҮҚиҰҒдәӢйЎ№пјҡдёҠйқўиҝҷеҸҘиҜқж„Ҹе‘ізқҖпјҢXML ж–Үд»¶еҸҜдҪҝз”ЁеӨҡдёӘдёҚеҗҢзҡ„ XSL ж ·ејҸиЎЁжқҘиҝӣиЎҢиҪ¬жҚўгҖӮ

еңЁжөҸи§ҲеҷЁдёӯжҠҠ XML иҪ¬жҚўдёә XHTML

иҝҷжҳҜз”ЁдәҺеңЁе®ўжҲ·з«ҜжҠҠ XML ж–Үд»¶иҪ¬жҚўдёә XHTML зҡ„жәҗд»Јз Ғпјҡ

<html>
<body>

<script type="text/javascript">

// Load XML 
var xml = new ActiveXObject("Microsoft.XMLDOM")
xml.async = false
xml.load("cdcatalog.xml")

// Load XSL
var xsl = new ActiveXObject("Microsoft.XMLDOM")
xsl.async = false
xsl.load("cdcatalog.xsl")

// Transform
document.write(xml.transformNode(xsl))

</script>

</body>
</html>

жҸҗзӨәпјҡеҒҮеҰӮжӮЁдёҚдәҶи§ЈеҰӮдҪ•зј–еҶҷ JavaScriptпјҢиҜ·еӯҰд№ жҲ‘д»¬зҡ„гҖҠJavaScript ж•ҷзЁӢгҖӢгҖӮ

з¬¬дёҖж®өд»Јз ҒеҲӣе»әдәҶеҫ®иҪҜзҡ„ XML и§ЈжһҗеҷЁзҡ„дёҖдёӘе®һдҫӢпјҢз„¶еҗҺжҠҠ XML ж–Үд»¶иҪҪе…ҘдәҶеҶ…еӯҳгҖӮз¬¬дәҢж®өд»Јз ҒеҲӣе»әдәҶи§ЈжһҗеҷЁзҡ„еҸҰдёҖдёӘе®һдҫӢпјҢз„¶еҗҺжҠҠиҝҷдёӘ XSL ж–Үд»¶иҪҪе…ҘдәҶеҶ…еӯҳгҖӮжңҖеҗҺдёҖиЎҢд»Јз ҒдҪҝз”Ё XSL ж–ҮжЎЈиҪ¬жҚўдәҶ XML ж–ҮжЎЈпјҢе№¶еңЁжөҸи§ҲеҷЁдёӯжҠҠз»“жһңдҪңдёә XHTML жҳҫзӨәеҮәжқҘгҖӮд»»еҠЎе®ҢжҲҗпјҒ
